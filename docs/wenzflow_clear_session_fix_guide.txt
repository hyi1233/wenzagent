  /// 清空当前会话
  Future<void> clearCurrentSession() async {
    try {
      final deviceClient = await DeviceClientFactory.getInstance(SpaceUtil.getCurrentSpaceId());
      if (deviceClient == null) return;

      // 使用 AgentProxy 清除会话（同时清除内存和数据库）
      if (_agentProxy != null) {
        await _agentProxy.clearCurrentSession();
        AILogger.debug('【清除会话】通过 AgentProxy 清除会话成功', extra: {'employeeId': employeeId});
      } else {
        // 降级方案：如果 AgentProxy 未初始化，只清除数据库
        AILogger.warning('【清除会话】AgentProxy 未初始化，使用降级方案清除数据库', extra: {'employeeId': employeeId});
        await deviceClient.messageStore.deleteMessages(employeeId);
      }

      // 清空 UI 缓存
      _messages.clear();
      _groupedMessages = const GroupedMessages();
      _permissionDecisions.clear();
      
      // 清理 Markdown 控制器缓存
      for (final controller in _markdownControllers.values) {
        controller.dispose();
      }
      _markdownControllers.clear();
      
      updateView();
      
      AILogger.debug('【清除会话】UI 缓存已清空', extra: {'employeeId': employeeId});
    } catch (e, stackTrace) {
      AILogger.error('清空会话失败', error: e, stackTrace: stackTrace);
    }
  }

问题修复说明
================================================================

修复前的问题：
- messageStore.deleteMessages() 只清除数据库
- Agent 内存中的消息未清除
- 下次 _loadMessages() 时，旧消息会重新出现

修复后：
- _agentProxy.clearCurrentSession() 同时清除：
  1. Agent 内存缓存 (PersistentChatAdapter.clearCurrentSession)
  2. 数据库消息 (通过 deleteMessagesCallback)
  3. 已持久化消息 ID 集合 (_persistedMessageIds)
- 保证内存和数据库的完全同步
